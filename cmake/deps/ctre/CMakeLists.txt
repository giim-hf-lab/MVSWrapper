FetchContent_Declare(ctre
	GIT_REPOSITORY "$CACHE{__C_GITHUB_PREFIX}hanickadot/compile-time-regular-expressions.git"

	GIT_SUBMODULES_RECURSE TRUE

	PROPERTY SYSTEM TRUE
